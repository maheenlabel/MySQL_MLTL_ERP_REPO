CREATE DEFINER=`root`@`%` PROCEDURE `InsertUpdateOffset_Technical_proc`(
IN Techonical_ID int,
IN SampleID int,
IN Overall_Length DECIMAL(18,5) ,
IN Overall_Width DECIMAL(18,5),
IN Finished_Length DECIMAL(18,5),
IN Finished_Width DECIMAL(18,5),
IN Fornt_Side_Color int,
IN Back_Side_Color int,
IN Cutting_Emboss int,
IN Cutting_Debossing int,
IN Die_Cutting int,
IN Punch_hole_size int,
IN Perforation int,
IN KeyEntry1 nvarchar(45),
IN KeyEntry2 nvarchar(45),
IN KeyEntry3 nvarchar(45),
IN KeyEntry4 nvarchar(45),
IN KeyEntry5 nvarchar(45),
IN KeyEntry6 nvarchar(45),
IN KeyEntry7 nvarchar(45),
IN KeyEntry8 nvarchar(45),
IN KeyEntry9 nvarchar(45),
IN KeyEntry10 nvarchar(45),
IN CreatedBy int,
IN CreationDate datetime,
IN Status int,
IN CommandID int
)
BEGIN
IF CommandID=1 THEN

INSERT INTO `npd_Offset_technical_tbl`
(
`SampleID`,
`Overall_Length`,
`Overall_Width`,
`Finished_Length`,
`Finished_Width`,
`Fornt_Side_Color`,
`Back_Side_Color`,
`Cutting_Emboss`,
`Cutting_Debossing`,
`Die_Cutting`,
`Punch_hole_size`,
`Perforation`,
`KeyEntry1`,
`KeyEntry2`,
`KeyEntry3`,
`KeyEntry4`,
`KeyEntry5`,
`KeyEntry6`,
`KeyEntry7`,
`KeyEntry8`,
`KeyEntry9`,
`KeyEntry10`,
`CreatedBy` ,
`CreationDate`,
`Status`)
VALUES
(
SampleID,
Overall_Length,
Overall_Width,
Finished_Length,
Finished_Width,
Fornt_Side_Color,
Back_Side_Color,
Cutting_Emboss,
Cutting_Debossing,
Die_Cutting,
Punch_hole_size,
Perforation,
KeyEntry1,
KeyEntry2,
KeyEntry3,
KeyEntry4,
KeyEntry5,
KeyEntry6,
KeyEntry7,
KeyEntry8,
KeyEntry9,
KeyEntry10,
 CreatedBy ,
CreationDate,
 Status);
ELSE 
UPDATE `npd_Offset_technical_tbl`
SET

`SampleID` = SampleID,
`Overall_Length` = Overall_Length,
`Overall_Width` = Overall_Width,
`Finished_Length` = Finished_Length,
`Finished_Width` = Finished_Width,
`Fornt_Side_Color` = Fornt_Side_Color,
`Back_Side_Color` = Back_Side_Color,
`Cutting_Emboss` = Cutting_Emboss,
`Cutting_Debossing` = Cutting_Debossing,
`Die_Cutting` = Die_Cutting,
`Punch_hole_size` = Punch_hole_size,
`Perforation` = Perforation,
`KeyEntry1` = KeyEntry1,
`KeyEntry2` = KeyEntry2,
`KeyEntry3` = KeyEntry3,
`KeyEntry4` = KeyEntry4,
`KeyEntry5` = KeyEntry5,
`KeyEntry6` = KeyEntry6,
`KeyEntry7` = KeyEntry7,
`KeyEntry8` = KeyEntry8,
`KeyEntry9` = KeyEntry9,
`KeyEntry10` = KeyEntry10,
 CreatedBy= CreatedBy ,
CreationDate=CreationDate,
 Status=Status

WHERE Offset_Technical_ID = Techonical_ID;
END IF;
END
