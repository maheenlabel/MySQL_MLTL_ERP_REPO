CREATE DEFINER=`root`@`%` PROCEDURE `InsertUpdatePFL_Techonical_proc`(
IN Techonical_ID int,
IN SampleID int,
IN Overall_Length DECIMAL(18,5) ,
IN Overall_Width DECIMAL(18,5),
IN Finished_Length DECIMAL(18,5),
IN Finished_Width DECIMAL(18,5),
IN Fornt_Side_Color int,
IN Back_Side_Color int,
IN Ribbon_Type nvarchar(45),
IN Ribbon_Color int,
IN Ribbon_Width DECIMAL(18,5),
IN Edge_Quality int,
IN Face_Quality int,
IN Item_Description nvarchar(45),
IN MachineID int,
IN FirstHeatPlateTemp int,
IN SecondHeatPlateTemp int,
 IN MachineSpeed nvarchar(45),
IN FinishType nvarchar(45),
IN FoldingType nvarchar(45),
IN CuringTime nvarchar(45),
IN CuiringTemparature nvarchar(45),
IN ItemAttributes nvarchar(45),
IN KeyEntry1 nvarchar(45),
IN KeyEntry2 nvarchar(45),
IN KeyEntry3 nvarchar(45),
IN KeyEntry4 nvarchar(45),
IN KeyEntry5 nvarchar(45),
IN KeyEntry6 nvarchar(45),
IN KeyEntry7 nvarchar(45),
IN KeyEntry8 nvarchar(45),
IN KeyEntry9 nvarchar(45),
IN KeyEntry10 nvarchar(45),
IN CreatedBy int,
IN CreationDate datetime,
IN Status int,
IN CommandID int
)
BEGIN
IF CommandID=1 THEN

INSERT INTO `npd_pfl_techonical_tbl`
(
`SampleID`,
`Overall_Length`,
`Overall_Width`,
`Finished_Length`,
`Finished_Width`,
`Fornt_Side_Color`,
`Back_Side_Color`,
`Ribbon_Type`,
`Ribbon_Color`,
`Ribbon_Width`,
`Edge_Quality`,
`Face_Quality`,
`Item_Description`,
`MachineID`,
`FirstHeatPlateTemp`,
`SecondHeatPlateTemp`,
`MachineSpeed`,
`FinishType`,
`FoldingType`,
`CuringTime`,
`CuiringTemparature`,
`ItemAttributes`,
`KeyEntry1`,
`KeyEntry2`,
`KeyEntry3`,
`KeyEntry4`,
`KeyEntry5`,
`KeyEntry6`,
`KeyEntry7`,
`KeyEntry8`,
`KeyEntry9`,
`KeyEntry10`,
`CreatedBy` ,
`CreationDate`,
`Status`)
VALUES
(
SampleID,
Overall_Length,
Overall_Width,
Finished_Length,
Finished_Width,
Fornt_Side_Color,
Back_Side_Color,
Ribbon_Type,
Ribbon_Color,
Ribbon_Width,
Edge_Quality,
Face_Quality,
Item_Description,
MachineID,
FirstHeatPlateTemp,
SecondHeatPlateTemp,
MachineSpeed,
FinishType,
FoldingType,
CuringTime,
CuiringTemparature,
ItemAttributes,
KeyEntry1,
KeyEntry2,
KeyEntry3,
KeyEntry4,
KeyEntry5,
KeyEntry6,
KeyEntry7,
KeyEntry8,
KeyEntry9,
KeyEntry10,
 CreatedBy ,
CreationDate,
 Status);
ELSE IF CommandID=2 THEN
UPDATE `npd_pfl_techonical_tbl`
SET

`SampleID` = SampleID,
`Overall_Length` = Overall_Length,
`Overall_Width` = Overall_Width,
`Finished_Length` = Finished_Length,
`Finished_Width` = Finished_Width,
`Fornt_Side_Color` = Fornt_Side_Color,
`Back_Side_Color` = Back_Side_Color,
`Ribbon_Type` = Ribbon_Type,
`Ribbon_Color` = Ribbon_Color,
`Ribbon_Width` = Ribbon_Width,
`Edge_Quality` = Edge_Quality,
`Face_Quality` = Face_Quality,
`Item_Description` = Item_Description,
`MachineID` = MachineID,
`FirstHeatPlateTemp` = MachineID,
`SecondHeatPlateTemp` = SecondHeatPlateTemp,
`MachineSpeed` = MachineSpeed,
`FinishType` = FinishType,
`FoldingType` = FoldingType,
`CuringTime` = CuringTime,
`CuiringTemparature` = CuiringTemparature,
`ItemAttributes` = ItemAttributes,
`KeyEntry1` = KeyEntry1,
`KeyEntry2` = KeyEntry2,
`KeyEntry3` = KeyEntry3,
`KeyEntry4` = KeyEntry4,
`KeyEntry5` = KeyEntry5,
`KeyEntry6` = KeyEntry6,
`KeyEntry7` = KeyEntry7,
`KeyEntry8` = KeyEntry8,
`KeyEntry9` = KeyEntry9,
`KeyEntry10` = KeyEntry10,
 CreatedBy= CreatedBy ,
CreationDate=CreationDate,
 Status=Status

WHERE `PFL_Techonical_ID` = Techonical_ID;
END IF;
END IF;
END
