CREATE DEFINER=`root`@`%` PROCEDURE `InsertUpdateNpd_Woven_Spec`(
in woven_ID int,
in SampleID int,
in SampleName nvarchar(45),
in BrandID int,
in ProgramID int,
in ProductCategoryID int,
in ProductSubCategoryID int,
in ClientID int,
in CompanyID int,
in NPDExecutive int,
in Unit int,
in Length decimal(18,5),
in Width decimal(18,5),
in QuotedPrice nvarchar(45),
in ArtworkLocation_FileName nvarchar(45),
in Imagepath nvarchar(45),
 in Cancel int,
in CreatedBy int,
 in CreationDate datetime,
in Status int,
in TotalNoOfColor int,
in ModifyBy int,
in ModifyDate datetime,
IN CommandID int
)
BEGIN
If CommandID=1 then
INSERT INTO `npd_woven_spec_tbl`
(`SampleID`,
`SampleName`,
`BrandID`,
`ProgramID`,
`ProductCategoryID`,
`ProductSubCategoryID`,
`ClientID`,
`CompanyID`,
`NPDExecutive`,
`Unit`,
`Length`,
`Width`,
`QuotedPrice`,
`ArtworkLocation_FileName`,
`Imagepath`,
`Cancel`,
`CreatedBy`,
`CreationDate`,
`Status`,
`TotalNoOfColor`,
`ModifyBy`,
`ModifyDate`)
VALUES
(SampleID,
SampleName,
BrandID,
ProgramID,
ProductCategoryID,
ProductSubCategoryID,
ClientID,
CompanyID,
NPDExecutive,
Unit,
Length,
Width,
QuotedPrice,
ArtworkLocation_FileName,
Imagepath,
Cancel,
CreatedBy,
CreationDate,
Status,
TotalNoOfColor,
ModifyBy,
ModifyDate);
else
UPDATE `mltl-erp-latest`.`npd_woven_spec_tbl`
SET
`SampleID` = SampleID,
`SampleName`=SampleName,
`BrandID` = BrandID,
`ProgramID` = ProgramID,
`ProductCategoryID` = ProductCategoryID,
`ProductSubCategoryID` = ProductSubCategoryID,
`ClientID` = ClientID,
`CompanyID` = CompanyID,
`NPDExecutive` = NPDExecutive,
`Unit` = Unit,
`Length` = Length,
`Width` = Width,
`QuotedPrice` = QuotedPrice,
`ArtworkLocation_FileName` = ArtworkLocation_FileName,
`Imagepath` = Imagepath,
`Cancel` = Cancel,
`CreatedBy` = CreatedBy,
`CreationDate` = CreationDate,
`Status` = Status,
`TotalNoOfColor` = TotalNoOfColor,
`ModifyBy` = ModifyBy,
`ModifyDate` = ModifyDate
WHERE `npd_woven_ID` = woven_ID;
end if;

END
